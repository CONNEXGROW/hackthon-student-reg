saveEditBtn.addEventListener("click", async () => { const photoInput = document.getElementById("editPhoto"); const photoFile = photoInput.files[0]; if (!photoFile) { showStatus(dashboardStatus, "Please upload a student photo.", true); return; } // Validate file type const allowedTypes = ["image/jpeg", "image/png", "image/jpg"]; if (!allowedTypes.includes(photoFile.type)) { showStatus(dashboardStatus, "Only JPG, PNG, or JPEG files are allowed.", true); return; } try { // Upload photo to Firebase Storage const photoURL = await uploadPhoto(photoFile); const updatedData = { fullName: document.getElementById("editFullName").value, collegeName: document.getElementById("editCollegeName").value, course: document.getElementById("editCourse").value, currentYear: parseInt(document.getElementById("editCurrentYear").value), specialization: document.getElementById("editSpecialization").value || null, mobileNumber: document.getElementById("editMobileNumber").value, emailId: document.getElementById("editEmailId").value, photo: photoURL, }; const response = await fetch( `https://hackthon-ujtu.onrender.com/api/admin/students/${currentEditId}`, { method: "PUT", headers: { "Content-Type": "application/json", Authorization: `Bearer ${adminToken}`, }, body: JSON.stringify(updatedData), } ); if (!response.ok) { const errorData = await response.json(); throw new Error(errorData.error || "Failed to update student"); } showStatus(dashboardStatus, "Student updated successfully!", false); editModal.style.display = "none"; fetchStudents(); } catch (error) { showStatus(dashboardStatus, "Error: " + error.message, true); console.error("Update error:", error); } });
